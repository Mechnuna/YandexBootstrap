Для привязки выключателя к компоненту с выбором часового пояса добавьте в тег часового пояса id

Например id="time_select"и в select еще один id

Например id="time_form"
(Каждый id должен быть уникальным) и в input тогла добавьте onclick="toggle('первый_id','второй_id')"

        
          <div class="toogle_block">
          <h2>Часовой пояс</h2>

          <div class="container-toogle">
            <input type="checkbox" name="toggle" class="toggle-button"
           onclick="toggle('time_form','time_select')">
            <span class="toogle_label">Определять время автоматически</span>
          </div>
        </div>


        <div class="time_select time_select_off" id="time_select">
          <label for="country" class="form_label">Часовой пояс</label>
          <select class="form-select form-select-off" id="time_form"required="">
            <option value="">(GMT +03.00) Москва, Санкт-Петербург</option>
            <option>(GMT +02.00) Калининград, Свердловск</option>
          </select>
        </div>
